CREATE TABLE Account(
   id  SERIAL PRIMARY KEY,
   email         character(250)      NOT NULL,
   password            character(250)      NOT NULL,
   role        integer,
   status         boolean
);

CREATE TABLE Product(
   id  SERIAL PRIMARY KEY,
   judul         character(250)      NOT NULL,
   fotoProduk            character(250)      ,
   id_merchant        integer,
   kategori         character(250),
   deskripsi		text,
	tahunTerbit		character(250),
	harga			integer,
	penulis			character(250),
	ISBN			character(100),
	penerbit		character(250)
	
);

CREATE TABLE Library(
   id  SERIAL PRIMARY KEY,
   customerId         integer      NOT NULL,
   productId            integer      NOT NULL,
   tanggal        date
);

CREATE TABLE Cart(
   id  SERIAL PRIMARY KEY,

CREATE TABLE Wishlist(
   id  SERIAL PRIMARY KEY,

CREATE TABLE Customer(
   id  SERIAL PRIMARY KEY,
   accountId         integer      NOT NULL,
   gambar            character(300),
   telepon        character(30),
	nama		character(300),
	alamat			character(300),
	tanggalLahir	date
);

CREATE TABLE Merchant(
   id  SERIAL PRIMARY KEY,

   namaToko		character(300),	
   gambar            character(300),
   telepon        character(30),
   nama		character(300),
   alamat			character(300),
   tanggalLahir	date
